#!/bin/bash
FROM=iso-8859-1
TO=utf-8
ICONV="iconv -f $FROM -t $TO"

# Convert
find ./ -type f -name "*.srt" | while read fn; do

    IS_TARGET=`file "${fn}" | grep -i iso-8859`
    
    if [ "$IS_TARGET" = "" ]; then

        echo "${fn} ---- Will NOT be converted!"

    else
        
        echo "${fn} ---- Will be converted!"
        cp "${fn}" "${fn}.bak"
        $ICONV < "${fn}.bak" > "${fn}"
        
    fi
done
